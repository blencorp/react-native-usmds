## Usage

```tsx
import { Text } from '@/components/ui/text';

<Text>Sample Text</Text>
```

## Props

### Text

Extends React Native's `Text` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| variant | `'default' \| 'h1' \| 'h2' \| 'h3' \| 'h4' \| 'p' \| 'blockquote' \| 'code' \| 'lead' \| 'large' \| 'small' \| 'muted'` | `'default'` | The text variant style. |
| asChild | `boolean` | `false` | Use Slot component instead of Text. |
| className | `string` | - | Additional classes for styling. |

## Variants

### Default

```tsx
<Text>Default Text</Text>
```

### Headings

```tsx
import { Text } from '@/components/ui/text';
import { View } from 'react-native';

<View className="gap-4">
  <Text variant="h1">Heading 1</Text>
  <Text variant="h2">Heading 2</Text>
  <Text variant="h3">Heading 3</Text>
  <Text variant="h4">Heading 4</Text>
</View>
```

### Paragraph

```tsx
<Text variant="p">
  This is a paragraph with proper spacing and line height for improved readability.
</Text>
```

### Blockquote

```tsx
<Text variant="blockquote">
  "This is a blockquote with italic styling and a left border."
</Text>
```

### Code

```tsx
<Text variant="code">const hello = "world";</Text>
```

### Lead

```tsx
<Text variant="lead">
  This is lead text, typically used for introductory paragraphs.
</Text>
```

### Large

```tsx
<Text variant="large">Large Text</Text>
```

### Small

```tsx
<Text variant="small">Small Text</Text>
```

### Muted

```tsx
<Text variant="muted">Muted text with reduced emphasis.</Text>
```


## Helper Functions

### textVariants

Use `textVariants` to create custom text-styled components:

```tsx
import { textVariants } from '@/components/ui/text';
import { Text as RNText } from 'react-native';

<RNText className={textVariants({ variant: 'h1' })}>
  Custom Text Component
</RNText>
```
