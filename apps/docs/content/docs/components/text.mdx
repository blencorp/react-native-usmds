## Usage

```tsx
import { Text } from '@/components/ui/text';

<Text>Sample Text</Text>
```

## Props

### Text

Extends React Native's `Text` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| variant | `'default' \| 'h1' \| 'h2' \| 'h3' \| 'h4' \| 'p' \| 'blockquote' \| 'code' \| 'lead' \| 'large' \| 'small' \| 'muted'` | `'default'` | The text variant style. |
| asChild | `boolean` | `false` | Use Slot component instead of Text. |
| className | `string` | - | Additional classes for styling. |

## Variants

### Default

```tsx
<Text>Default Text</Text>
```

### Headings

```tsx
import { Text } from '@/components/ui/text';
import { View } from 'react-native';

<View className="gap-4">
  <Text variant="h1">Heading 1</Text>
  <Text variant="h2">Heading 2</Text>
  <Text variant="h3">Heading 3</Text>
  <Text variant="h4">Heading 4</Text>
</View>
```

### Paragraph

```tsx
<Text variant="p">
  This is a paragraph with proper spacing and line height for improved readability.
</Text>
```

### Blockquote

```tsx
<Text variant="blockquote">
  "This is a blockquote with italic styling and a left border."
</Text>
```

### Code

```tsx
<Text variant="code">const hello = "world";</Text>
```

### Lead

```tsx
<Text variant="lead">
  This is lead text, typically used for introductory paragraphs.
</Text>
```

### Large

```tsx
<Text variant="large">Large Text</Text>
```

### Small

```tsx
<Text variant="small">Small Text</Text>
```

### Muted

```tsx
<Text variant="muted">Muted text with reduced emphasis.</Text>
```

### Typography Showcase

```tsx
import { Text } from '@/components/ui/text';
import { View } from 'react-native';

function TypographyShowcase() {
  return (
    <View className="gap-6 p-4">
      <Text variant="h1">Typography System</Text>

      <Text variant="lead">
        A comprehensive text component with multiple variants for different use cases.
      </Text>

      <View className="gap-4">
        <Text variant="h2">Section Title</Text>
        <Text variant="p">
          This is a regular paragraph with default styling. It maintains good readability
          with proper line height and spacing.
        </Text>

        <Text variant="h3">Subsection</Text>
        <Text variant="muted">
          This is muted text, often used for secondary information or descriptions.
        </Text>
      </View>
    </View>
  );
}
```

### With TextClassContext

```tsx
import { Text, TextClassContext } from '@/components/ui/text';
import { View } from 'react-native';

function StyledSection() {
  return (
    <TextClassContext.Provider value="text-primary">
      <View className="gap-2">
        <Text>This text inherits primary color</Text>
        <Text>So does this one</Text>
        <Text className="text-secondary">But this overrides it</Text>
      </View>
    </TextClassContext.Provider>
  );
}
```

### List with Text Variants

```tsx
function FeatureList() {
  const features = [
    'Fast performance',
    'Easy to use',
    'Fully customizable',
    'Open source'
  ];

  return (
    <View className="gap-4">
      <Text variant="h3">Features</Text>
      <View className="gap-2">
        {features.map((feature, index) => (
          <Text key={index} variant="p">
            {index + 1}. {feature}
          </Text>
        ))}
      </View>
    </View>
  );
}
```

## Helper Functions

### textVariants

Use `textVariants` to create custom text-styled components:

```tsx
import { textVariants } from '@/components/ui/text';
import { Text as RNText } from 'react-native';

<RNText className={textVariants({ variant: 'h1' })}>
  Custom Text Component
</RNText>
```
