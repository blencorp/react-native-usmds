## Usage

```tsx
import {
  Collapsible,
  CollapsibleTrigger,
  CollapsibleContent,
} from '@/components/ui/collapsible';
import { useState } from 'react';

const [isOpen, setIsOpen] = useState(false);

<Collapsible open={isOpen} onOpenChange={setIsOpen}>
  <CollapsibleTrigger asChild>
    <Button variant="outline">
      <Text>Toggle</Text>
    </Button>
  </CollapsibleTrigger>
  <CollapsibleContent>
    <Text>Content here</Text>
  </CollapsibleContent>
</Collapsible>
```

## Props

### Collapsible

Extends `@rn-primitives/collapsible` Root props.

| Prop | Type | Description |
|------|------|-------------|
| open | `boolean` | The controlled open state. |
| onOpenChange | `(open: boolean) => void` | Callback when open state changes. |
| defaultOpen | `boolean` | The default open state when initially rendered. |
| disabled | `boolean` | Whether the collapsible is disabled. |

### CollapsibleTrigger

Extends `@rn-primitives/collapsible` Trigger props.

| Prop | Type | Description |
|------|------|-------------|
| asChild | `boolean` | Render as child component (uses Slot). |

### CollapsibleContent

Extends `@rn-primitives/collapsible` Content props.

| Prop | Type | Description |
|------|------|-------------|
| forceMount | `boolean` | Force mount when more control is needed. |

## Examples

### Default

```tsx
import {
  Collapsible,
  CollapsibleTrigger,
  CollapsibleContent,
} from '@/components/ui/collapsible';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { useState } from 'react';

const [isOpen, setIsOpen] = useState(false);

<Collapsible open={isOpen} onOpenChange={setIsOpen}>
  <CollapsibleTrigger asChild>
    <Button variant="outline">
      <Text>Toggle Content</Text>
    </Button>
  </CollapsibleTrigger>
  <CollapsibleContent>
    <View className="p-4">
      <Text>This content can be expanded and collapsed.</Text>
    </View>
  </CollapsibleContent>
</Collapsible>
```

### With Icon

```tsx
import { ChevronDown } from 'lucide-react-native';

<Collapsible open={isOpen} onOpenChange={setIsOpen}>
  <CollapsibleTrigger asChild>
    <Button variant="outline">
      <Text>Details</Text>
      <ChevronDown className={isOpen ? 'rotate-180' : ''} />
    </Button>
  </CollapsibleTrigger>
  <CollapsibleContent>
    <View className="p-4">
      <Text>Additional details here</Text>
    </View>
  </CollapsibleContent>
</Collapsible>
```
