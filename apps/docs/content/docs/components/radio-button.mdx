## Usage

```tsx
import { RadioGroup, RadioButton } from '@/components/ui/radiobutton';
import { useState } from 'react';

const [value, setValue] = useState('1');

<RadioGroup value={value} onValueChange={setValue}>
  <RadioButton value="1" label="Option 1" />
  <RadioButton value="2" label="Option 2" />
  <RadioButton value="3" label="Option 3" />
</RadioGroup>
```

## Props

### RadioButton

Extends `RadioGroupPrimitive.Item` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| label | `string` | - | **Required.** The text label for the radio button. |
| value | `string` | - | **Required.** The value associated with this radio button. |
| disabled | `boolean` | `false` | When true, prevents user interaction. |
| className | `string` | - | Additional classes for styling. |

### RadioGroup

Extends `RadioGroupPrimitive.Root` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | `string` | - | The value of the selected radio button. |
| onValueChange | `(value: string) => void` | - | Callback fired when selection changes. |

## Examples

### Default

```tsx
import { RadioGroup, RadioButton } from '@/components/ui/radiobutton';
import { useState } from 'react';

function RadioButtonExample() {
  const [value, setValue] = useState('1');

  return (
    <RadioGroup value={value} onValueChange={setValue}>
      <RadioButton value="1" label="Option 1" />
      <RadioButton value="2" label="Option 2" />
      <RadioButton value="3" label="Option 3" />
    </RadioGroup>
  );
}
```

### Single Radio Button

```tsx
<RadioGroup value={value} onValueChange={setValue}>
  <RadioButton value="1" label="Single option" />
</RadioGroup>
```

### Disabled Options

```tsx
<RadioGroup value="2" onValueChange={setValue}>
  <RadioButton value="1" label="Disabled unselected" disabled />
  <RadioButton value="2" label="Selected" />
  <RadioButton value="3" label="Disabled selected" disabled />
</RadioGroup>
```

### With Form

```tsx
import { RadioGroup, RadioButton } from '@/components/ui/radiobutton';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { useState } from 'react';
import { View } from 'react-native';

function RadioForm() {
  const [size, setSize] = useState('medium');

  const handleSubmit = () => {
    console.log('Selected size:', size);
  };

  return (
    <View className="gap-4">
      <Text className="font-semibold">Choose a size</Text>
      <RadioGroup value={size} onValueChange={setSize}>
        <RadioButton value="small" label="Small" />
        <RadioButton value="medium" label="Medium" />
        <RadioButton value="large" label="Large" />
        <RadioButton value="xlarge" label="Extra Large" />
      </RadioGroup>
      <Button onPress={handleSubmit}>
        <Text>Submit</Text>
      </Button>
    </View>
  );
}
```

### Controlled State

```tsx
function ControlledRadio() {
  const [value, setValue] = useState('option1');

  const handleChange = (newValue: string) => {
    console.log('Changed to:', newValue);
    setValue(newValue);
  };

  return (
    <RadioGroup value={value} onValueChange={handleChange}>
      <RadioButton value="option1" label="First option" />
      <RadioButton value="option2" label="Second option" />
      <RadioButton value="option3" label="Third option" />
    </RadioGroup>
  );
}
```
