## Usage

```tsx

import { useState } from 'react';

const [text, setText] = useState('');

<TextInput
  placeholder="Enter text..."
  value={text}
  onChangeText={setText}
/>
```

## Props

### TextInput

Extends React Native's `TextInput` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| className | `string` | - | Additional classes for styling. |
| editable | `boolean` | `true` | When false, input is disabled. |
| placeholder | `string` | - | Placeholder text. |
| value | `string` | - | The input value. |
| onChangeText | `(text: string) => void` | - | Callback when text changes. |

All standard React Native TextInput props are supported.

### With Error State

```tsx
import { TextInput } from '@/components/ui/textinput';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';
import { useState } from 'react';

function InputWithError() {
  const [email, setEmail] = useState('');
  const [error, setError] = useState('');

  const validateEmail = (text: string) => {
    setEmail(text);
    if (text && !text.includes('@')) {
      setError('Please enter a valid email');
    } else {
      setError('');
    }
  };

  return (
    <View className="gap-2">
      <TextInput
        placeholder="Email"
        value={email}
        onChangeText={validateEmail}
        aria-invalid={!!error}
        className={error ? 'border-destructive' : ''}
      />
      {error && <Text className="text-sm text-destructive">{error}</Text>}
    </View>
  );
}
```

### Form Example

```tsx
import { TextInput } from '@/components/ui/textinput';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';
import { useState } from 'react';

function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');

  const handleSubmit = () => {
    console.log('Email:', email);
    console.log('Password:', password);
  };

  return (
    <View className="gap-4">
      <View className="gap-2">
        <Text className="font-medium">Email</Text>
        <TextInput
          placeholder="your@email.com"
          value={email}
          onChangeText={setEmail}
          keyboardType="email-address"
          autoCapitalize="none"
        />
      </View>

      <View className="gap-2">
        <Text className="font-medium">Password</Text>
        <TextInput
          placeholder="Enter password"
          value={password}
          onChangeText={setPassword}
          secureTextEntry
        />
      </View>

      <Button onPress={handleSubmit}>
        <Text>Login</Text>
      </Button>
    </View>
  );
}
```

### With Character Counter

```tsx
import { TextInput } from '@/components/ui/textinput';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';
import { useState } from 'react';

function InputWithCounter() {
  const [text, setText] = useState('');
  const maxLength = 100;

  return (
    <View className="gap-2">
      <TextInput
        placeholder="Enter text..."
        value={text}
        onChangeText={setText}
        maxLength={maxLength}
      />
      <Text className="text-sm text-muted-foreground text-right">
        {text.length}/{maxLength}
      </Text>
    </View>
  );
}
```
