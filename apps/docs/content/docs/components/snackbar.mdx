## Usage

```tsx
import { Snackbar } from '@/components/ui/snackbar';
import { useState } from 'react';

const [isVisible, setIsVisible] = useState(true);

<Snackbar
  isVisible={isVisible}
  variant="success"
  message="Message sent successfully"
  onAction={() => console.log('Action')}
  onDismiss={() => setIsVisible(false)}
/>
```

## Props

### Snackbar

Extends `View` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| isVisible | `boolean` | - | **Required.** Controls the visibility of the snackbar. |
| message | `string` | - | **Required.** The message to display in the snackbar. |
| variant | `'success' \| 'error'` | `'success'` | The visual style variant of the snackbar. |
| layout | `'one-line' \| 'two-lines'` | `'one-line'` | The layout style of the snackbar. |
| onAction | `() => void` | - | Optional callback when action button is clicked. |
| onDismiss | `() => void` | - | Optional callback when dismiss button is clicked. |
| actionLabel | `string` | `'Action'` | Text for the action button. |
| className | `string` | - | Additional classes for styling. |

### Auto Dismiss

```tsx
import { Snackbar } from '@/components/ui/snackbar';
import { useState, useEffect } from 'react';

function AutoDismissSnackbar() {
  const [isVisible, setIsVisible] = useState(false);

  const showSnackbar = () => {
    setIsVisible(true);
    setTimeout(() => {
      setIsVisible(false);
    }, 3000); // Auto dismiss after 3 seconds
  };

  return (
    <>
      <Button onPress={showSnackbar}>
        <Text>Show Snackbar</Text>
      </Button>
      <Snackbar
        isVisible={isVisible}
        variant="success"
        message="Action completed"
        onDismiss={() => setIsVisible(false)}
      />
    </>
  );
}
```

### Form Submission Feedback

```tsx
import { Snackbar } from '@/components/ui/snackbar';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { useState } from 'react';
import { View } from 'react-native';

function FormWithSnackbar() {
  const [snackbar, setSnackbar] = useState({ visible: false, variant: 'success', message: '' });

  const handleSubmit = async () => {
    try {
      // Submit form logic
      await submitForm();
      setSnackbar({
        visible: true,
        variant: 'success',
        message: 'Form submitted successfully'
      });
    } catch (error) {
      setSnackbar({
        visible: true,
        variant: 'error',
        message: 'Failed to submit form'
      });
    }

    // Auto dismiss after 4 seconds
    setTimeout(() => {
      setSnackbar(prev => ({ ...prev, visible: false }));
    }, 4000);
  };

  return (
    <View>
      <Button onPress={handleSubmit}>
        <Text>Submit Form</Text>
      </Button>
      <Snackbar
        isVisible={snackbar.visible}
        variant={snackbar.variant}
        message={snackbar.message}
        onDismiss={() => setSnackbar(prev => ({ ...prev, visible: false }))}
      />
    </View>
  );
}
```
