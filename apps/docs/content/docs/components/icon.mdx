## Usage

```tsx
import { Icon } from '@/components/ui/icon';
import { Home } from 'lucide-react-native';

<Icon as={Home} size={24} className="text-foreground" />
```

## Props

### Icon

Extends Lucide icon props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| as | `LucideIcon` | - | **Required.** The Lucide icon component to render. |
| size | `number` | `14` | The size of the icon. |
| className | `string` | `'text-foreground'` | Additional classes for styling. |

## Examples

### Default

```tsx
import { Icon } from '@/components/ui/icon';
import { Home } from 'lucide-react-native';

<Icon as={Home} />
```

### Custom Size

```tsx
import { Icon } from '@/components/ui/icon';
import { Settings } from 'lucide-react-native';

<Icon as={Settings} size={32} />
```

### Custom Color

```tsx
import { Icon } from '@/components/ui/icon';
import { Heart } from 'lucide-react-native';

<Icon as={Heart} className="text-destructive" />
```

### With Button

```tsx
import { Icon } from '@/components/ui/icon';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { Mail } from 'lucide-react-native';

<Button variant="default">
  <Icon as={Mail} size={16} />
  <Text>Send Email</Text>
</Button>
```

### Multiple Icons

```tsx
import { Icon } from '@/components/ui/icon';
import { Home, Settings, User, Bell } from 'lucide-react-native';
import { View } from 'react-native';

<View className="flex-row gap-4">
  <Icon as={Home} size={20} />
  <Icon as={Settings} size={20} />
  <Icon as={User} size={20} />
  <Icon as={Bell} size={20} />
</View>
```
