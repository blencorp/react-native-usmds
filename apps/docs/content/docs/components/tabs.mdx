## Usage

```tsx
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';
import { Text } from '@/components/ui/text';

<Tabs defaultValue="tab1">
  <TabsList>
    <TabsTrigger value="tab1">
      <Text>Tab 1</Text>
    </TabsTrigger>
    <TabsTrigger value="tab2">
      <Text>Tab 2</Text>
    </TabsTrigger>
  </TabsList>
  <TabsContent value="tab1">
    <Text>Content 1</Text>
  </TabsContent>
  <TabsContent value="tab2">
    <Text>Content 2</Text>
  </TabsContent>
</Tabs>
```

## Props

### Tabs

Extends `@rn-primitives/tabs` Root props.

| Prop | Type | Description |
|------|------|-------------|
| value | `string` | The controlled value of the tab to activate. |
| onValueChange | `(value: string) => void` | Event handler called when the value changes. |
| defaultValue | `string` | The value of the tab that should be active when initially rendered. |
| orientation | `'horizontal' \| 'vertical'` | The orientation of the tabs. Default is `'horizontal'`. |
| disabled | `boolean` | When true, prevents user interaction. |
| className | `string` | Additional classes for styling. |

### TabsList

| Prop | Type | Description |
|------|------|-------------|
| children | `React.ReactNode` | The tab triggers. |
| className | `string` | Additional classes for styling. |

### TabsTrigger

Extends `@rn-primitives/tabs` Trigger props.

| Prop | Type | Description |
|------|------|-------------|
| value | `string` | **Required.** A unique value that associates the trigger with a content. |
| disabled | `boolean` | When true, prevents the user from activating the tab. |
| className | `string` | Additional classes for styling. |

### TabsContent

Extends `@rn-primitives/tabs` Content props.

| Prop | Type | Description |
|------|------|-------------|
| value | `string` | **Required.** A unique value that associates the content with a trigger. |
| forceMount | `boolean` | Force mounting content even when not active. |
| className | `string` | Additional classes for styling. |

```tsx
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';
import { Text } from '@/components/ui/text';

function TabsExample() {
```

### Controlled Tabs

```tsx
import { useState } from 'react';

function ControlledTabsExample() {
  const [activeTab, setActiveTab] = useState('account');

  return (
    <Tabs value={activeTab} onValueChange={setActiveTab}>
      <TabsList>
        <TabsTrigger value="account">
          <Text>Account</Text>
        </TabsTrigger>
        <TabsTrigger value="password">
          <Text>Password</Text>
        </TabsTrigger>
      </TabsList>
      <TabsContent value="account">
        <Text>Account settings</Text>
      </TabsContent>
      <TabsContent value="password">
        <Text>Password settings</Text>
      </TabsContent>
    </Tabs>
  );
}
```
