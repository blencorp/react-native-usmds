## Usage

```tsx
import {
  Dialog,
  DialogTrigger,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogDescription,
} from '@/components/ui/dialog';

<Dialog>
  <DialogTrigger asChild>
    <Button variant="outline">
      <Text>Open</Text>
    </Button>
  </DialogTrigger>
  <DialogContent>
    <DialogHeader>
      <DialogTitle>Title</DialogTitle>
      <DialogDescription>Description</DialogDescription>
    </DialogHeader>
  </DialogContent>
</Dialog>
```

## Props

### Dialog

Extends `@rn-primitives/dialog` Root props.

| Prop | Type | Description |
|------|------|-------------|
| open | `boolean` | The controlled open state. |
| onOpenChange | `(open: boolean) => void` | Callback when open state changes. |
| defaultOpen | `boolean` | The default open state. |

### DialogContent

Extends `@rn-primitives/dialog` Content props.

| Prop | Type | Description |
|------|------|-------------|
| className | `string` | Additional classes for styling. |
| portalHost | `string` | Optional portal host name. |

### DialogHeader

Extends `View` props.

| Prop | Type | Description |
|------|------|-------------|
| className | `string` | Additional classes for styling. |

### DialogFooter

Extends `View` props.

| Prop | Type | Description |
|------|------|-------------|
| className | `string` | Additional classes for styling. |

### DialogTitle

Extends `@rn-primitives/dialog` Title props.

| Prop | Type | Description |
|------|------|-------------|
| className | `string` | Additional classes for styling. |

### DialogDescription

Extends `@rn-primitives/dialog` Description props.

| Prop | Type | Description |
|------|------|-------------|
| className | `string` | Additional classes for styling. |

## Examples

### Default

```tsx
import {
  Dialog,
  DialogTrigger,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogDescription,
} from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';

<Dialog>
  <DialogTrigger asChild>
    <Button variant="outline">
      <Text>Open Dialog</Text>
    </Button>
  </DialogTrigger>
  <DialogContent>
    <DialogHeader>
      <DialogTitle>Are you sure?</DialogTitle>
      <DialogDescription>
        This action cannot be undone.
      </DialogDescription>
    </DialogHeader>
  </DialogContent>
</Dialog>
```

### With Footer

```tsx
import { DialogFooter, DialogClose } from '@/components/ui/dialog';

<DialogContent>
  <DialogHeader>
    <DialogTitle>Confirm Action</DialogTitle>
    <DialogDescription>
      Please confirm you want to proceed.
    </DialogDescription>
  </DialogHeader>
  <DialogFooter>
    <DialogClose asChild>
      <Button variant="outline">
        <Text>Cancel</Text>
      </Button>
    </DialogClose>
    <DialogClose asChild>
      <Button variant="default">
        <Text>Confirm</Text>
      </Button>
    </DialogClose>
  </DialogFooter>
</DialogContent>
```
