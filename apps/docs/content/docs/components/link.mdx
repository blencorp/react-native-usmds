## Usage

```tsx
import { Link } from '@/components/ui/link';

<Link href="#" label="Link text" />
```

## Props

### Link

Extends `Pressable` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| href | `string` | - | **Required.** The URL the link points to. |
| label | `string` | - | **Required.** The text content of the link. |
| variant | `'default' \| 'visited' \| 'dark-background'` | `'default'` | The visual style variant. |
| external | `boolean` | `false` | Show external link icon. |
| visited | `boolean` | `false` | Apply visited link styling. |
| focus | `boolean` | `false` | Apply focus ring styling. |
| className | `string` | - | Additional classes for styling. |

## Variants

### Default

```tsx
<Link href="#" label="Default Link" variant="default" />
```

### Visited

```tsx
<Link href="#" label="Visited Link" visited />
```

### Dark Background

```tsx
<Link href="#" label="Dark Background Link" variant="dark-background" />
```


## Helper Functions

### linkVariants

Use `linkVariants` to create custom link-styled components:

```tsx
import { linkVariants } from '@/components/ui/link';
import { Text } from 'react-native';

<Text className={linkVariants({ variant: 'visited', external: true })}>
  Custom Link
</Text>
```
