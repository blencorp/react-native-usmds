## Usage

```tsx
import { RadioTileGroup, RadioTile } from '@/components/ui/radiotile';
import { useState } from 'react';

const [value, setValue] = useState('1');

<RadioTileGroup value={value} onValueChange={setValue}>
  <RadioTile value="1" label="Option 1" description="Description text" />
  <RadioTile value="2" label="Option 2" description="Description text" />
</RadioTileGroup>
```

## Props

### RadioTile

Extends `RadioGroupPrimitive.Item` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| label | `string` | - | **Required.** The text label for the radio tile. |
| value | `string` | - | **Required.** The value associated with this radio tile. |
| description | `string` | - | Optional description text displayed below the label. |
| variant | `'default' \| 'withDescription'` | Auto-detected | The layout variant (auto-set based on description presence). |
| disabled | `boolean` | `false` | When true, prevents user interaction. |
| className | `string` | - | Additional classes for styling. |

### RadioTileGroup

Extends `RadioGroupPrimitive.Root` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | `string` | - | The value of the selected radio tile. |
| onValueChange | `(value: string) => void` | - | Callback fired when selection changes. |

function RadioTileExample() {
  const [value, setValue] = useState('1');

    <RadioTileGroup value={value} onValueChange={setValue}>
      <RadioTile value="1" label="Option 1" />
      <RadioTile value="2" label="Option 2" />

## Helper Functions

### radioTileVariants

Use `radioTileVariants` to create custom radio tile-styled components:

```tsx
import { radioTileVariants } from '@/components/ui/radiotile';
import { View } from 'react-native';

<View className={radioTileVariants({ variant: 'withDescription', state: 'checked' })}>
  {/* custom content */}
</View>
```
