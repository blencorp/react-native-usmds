## Usage

```tsx
import {
  ContextMenu,
  ContextMenuContent,
  ContextMenuItem,
  ContextMenuTrigger,
} from '@/components/ui/context-menu';
import { Text } from '@/components/ui/text';

<ContextMenu>
  <ContextMenuTrigger>
    <Text>Right click or long press</Text>
  </ContextMenuTrigger>
  <ContextMenuContent>
    <ContextMenuItem>
      <Text>Edit</Text>
    </ContextMenuItem>
    <ContextMenuItem>
      <Text>Delete</Text>
    </ContextMenuItem>
  </ContextMenuContent>
</ContextMenu>
```

## Props

### ContextMenu

Extends `@rn-primitives/context-menu` Root props.

| Prop | Type | Description |
|------|------|-------------|
| open | `boolean` | The controlled open state of the context menu. |
| onOpenChange | `(open: boolean) => void` | Event handler called when the open state changes. |

### ContextMenuTrigger

Extends `@rn-primitives/context-menu` Trigger props.

| Prop | Type | Description |
|------|------|-------------|
| asChild | `boolean` | Change the default rendered element for the one passed as a child. |
| disabled | `boolean` | When true, prevents context menu from opening. |

### ContextMenuContent

Extends `@rn-primitives/context-menu` Content props.

| Prop | Type | Description |
|------|------|-------------|
| className | `string` | Additional classes for styling. |
| portalHost | `string` | The host where the context menu content will be rendered. |

### ContextMenuItem

Extends `@rn-primitives/context-menu` Item props.

| Prop | Type | Description |
|------|------|-------------|
| disabled | `boolean` | When true, prevents the user from interacting with the item. |
| textValue | `string` | Optional text used for typeahead purposes. |
| className | `string` | Additional classes for styling. |

## Examples

### Default

```tsx
import { Card } from '@/components/ui/card';

<ContextMenu>
  <ContextMenuTrigger asChild>
    <Card className="p-4">
      <Text>Right click or long press me</Text>
    </Card>
  </ContextMenuTrigger>
  <ContextMenuContent>
    <ContextMenuItem>
      <Text>Edit</Text>
    </ContextMenuItem>
    <ContextMenuItem>
      <Text>Duplicate</Text>
    </ContextMenuItem>
    <ContextMenuSeparator />
    <ContextMenuItem>
      <Text>Delete</Text>
    </ContextMenuItem>
  </ContextMenuContent>
</ContextMenu>
```

### With Icons

```tsx
import { Copy, Edit, Trash } from 'lucide-react-native';
import { Icon } from '@/components/ui/icon';

<ContextMenu>
  <ContextMenuTrigger asChild>
    <Card className="w-64 p-4">
      <Text>Right click for options</Text>
    </Card>
  </ContextMenuTrigger>
  <ContextMenuContent className="w-64">
    <ContextMenuItem>
      <View className="flex-row items-center gap-2">
        <Icon as={Edit} className="size-4" />
        <Text>Edit</Text>
      </View>
    </ContextMenuItem>
    <ContextMenuItem>
      <View className="flex-row items-center gap-2">
        <Icon as={Copy} className="size-4" />
        <Text>Duplicate</Text>
      </View>
    </ContextMenuItem>
    <ContextMenuSeparator />
    <ContextMenuItem>
      <View className="flex-row items-center gap-2">
        <Icon as={Trash} className="size-4 text-destructive" />
        <Text className="text-destructive">Delete</Text>
      </View>
    </ContextMenuItem>
  </ContextMenuContent>
</ContextMenu>
```
