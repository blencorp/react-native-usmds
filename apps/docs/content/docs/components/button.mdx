## Usage

```tsx
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';

<Button variant="default" onPress={() => console.log('pressed')}>
  <Text>Click me</Text>
</Button>
```

## Props

### Button

Extends `Pressable` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| variant | `'default' \| 'destructive' \| 'outline' \| 'secondary' \| 'ghost' \| 'link'` | `'default'` | The visual style variant. |
| size | `'default' \| 'sm' \| 'lg' \| 'icon'` | `'default'` | The size of the button. |
| className | `string` | - | Additional classes for styling. |
| disabled | `boolean` | `false` | Whether the button is disabled. |

## Helper Functions

### buttonVariants

Use `buttonVariants` to create button-styled components:

```tsx
import { buttonVariants } from '@/components/ui/button';
import { Pressable } from 'react-native';

<Pressable className={buttonVariants({ variant: 'outline', size: 'lg' })}>
  {/* content */}
</Pressable>
```

### buttonTextVariants

Use `buttonTextVariants` for text styling within buttons:

```tsx
import { buttonTextVariants } from '@/components/ui/button';
import { Text } from '@/components/ui/text';

<Text className={buttonTextVariants({ variant: 'destructive', size: 'sm' })}>
  Delete
</Text>
```
