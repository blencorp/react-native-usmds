## Usage

```tsx
import { Alert, AlertDescription, AlertTitle } from '@/components/ui/alert';
import { AlertCircle } from 'lucide-react-native';

<Alert icon={AlertCircle}>
  <AlertTitle>Heads up!</AlertTitle>
  <AlertDescription>
    You can add components to your app using the CLI.
  </AlertDescription>
</Alert>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| variant | `'default' \| 'destructive'` | `'default'` | The visual style variant |
| icon | `React.ComponentType` | - | Icon component to display |
| className | `string` | - | Additional styles |

## Examples

### Default Alert

```tsx
<Alert>
  <AlertTitle>Notification</AlertTitle>
  <AlertDescription>This is a default alert message.</AlertDescription>
</Alert>
```

### With Icon

```tsx
import { Info } from 'lucide-react-native';

<Alert icon={Info}>
  <AlertTitle>Information</AlertTitle>
  <AlertDescription>
    Here's some important information for you.
  </AlertDescription>
</Alert>
```

### Destructive Alert

```tsx
import { AlertCircle } from 'lucide-react-native';

<Alert variant="destructive" icon={AlertCircle}>
  <AlertTitle>Error</AlertTitle>
  <AlertDescription>
    Something went wrong. Please try again.
  </AlertDescription>
</Alert>
```

### Without Description

```tsx
<Alert icon={Terminal}>
  <AlertTitle>Command executed successfully</AlertTitle>
</Alert>
```

## Accessibility

- Alert components have appropriate ARIA roles
- Icons are decorative and marked with aria-hidden
- Color is not the only indicator of alert type
- Text contrast meets WCAG guidelines

## Best Practices

1. **Clear Messaging**: Use concise, actionable language
2. **Appropriate Variants**: Use destructive only for errors
3. **Icon Usage**: Include relevant icons to enhance meaning
4. **Positioning**: Place alerts where users will notice them
5. **Dismissible**: Consider making non-critical alerts dismissible
