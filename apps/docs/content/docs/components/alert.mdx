## Usage

```tsx
import { Alert, AlertTitle, AlertDescription } from '@/components/ui/alert';
import { AlertCircle } from 'lucide-react-native';

<Alert icon={AlertCircle} variant="default">
  <AlertTitle>Heads up!</AlertTitle>
  <AlertDescription>
    You can add components to your app using the CLI.
  </AlertDescription>
</Alert>
```

## Props

### Alert

Extends `View` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| icon | `LucideIcon` | - | **Required.** The icon component to display. |
| variant | `'default' \| 'destructive'` | `'default'` | The visual style variant. |
| iconClassName | `string` | - | Additional classes for the icon. |
| className | `string` | - | Additional classes for the container. |

### AlertTitle

Extends `Text` props.

| Prop | Type | Description |
|------|------|-------------|
| className | `string` | Additional classes for the title. |

### AlertDescription

Extends `Text` props.

| Prop | Type | Description |
|------|------|-------------|
| className | `string` | Additional classes for the description. |

## Examples

### Default

```tsx
import { Info } from 'lucide-react-native';

<Alert icon={Info} variant="default">
  <AlertTitle>Heads up!</AlertTitle>
  <AlertDescription>
    You can add components to your app using the CLI.
  </AlertDescription>
</Alert>
```

### Destructive

```tsx
import { AlertCircle } from 'lucide-react-native';

<Alert icon={AlertCircle} variant="destructive">
  <AlertTitle>Error</AlertTitle>
  <AlertDescription>
    Your session has expired. Please log in again.
  </AlertDescription>
</Alert>
```

### With Custom Icon

```tsx
import { Terminal } from 'lucide-react-native';

<Alert icon={Terminal} variant="default">
  <AlertTitle>System Update</AlertTitle>
  <AlertDescription>
    A new software update is available for your device.
  </AlertDescription>
</Alert>
```
