## Usage

```tsx
import { CheckboxTile } from '@/components/ui/checkbox-tile';
import { useState } from 'react';

const [checked, setChecked] = useState(false);

<CheckboxTile
  label="Checkbox tile"
  checked={checked}
  onCheckedChange={setChecked}
/>
```

## Props

### CheckboxTile

Extends `@rn-primitives/checkbox` Root props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| label | `string` | - | **Required.** The text label for the checkbox tile. |
| description | `string` | - | Optional description text below the label. |
| checked | `boolean` | - | The controlled checked state. |
| onCheckedChange | `(checked: boolean) => void` | - | Callback when checked state changes. |
| disabled | `boolean` | `false` | Whether the checkbox is disabled. |
| className | `string` | - | Additional classes for styling. |

## Examples

### Default

```tsx
import { CheckboxTile } from '@/components/ui/checkbox-tile';
import { useState } from 'react';

const [checked, setChecked] = useState(false);

<CheckboxTile
  label="Basic checkbox tile"
  checked={checked}
  onCheckedChange={setChecked}
/>
```

### With Description

```tsx
<CheckboxTile
  label="Checkbox with description"
  description="This is optional text that can be used to describe the label in more detail."
  checked={checked}
  onCheckedChange={setChecked}
/>
```

### Disabled

```tsx
<CheckboxTile
  label="Disabled checkbox"
  description="This checkbox is disabled"
  checked={false}
  disabled
  onCheckedChange={() => {}}
/>
```

### Disabled Checked

```tsx
<CheckboxTile
  label="Disabled checked"
  description="This checkbox is disabled and checked"
  checked={true}
  disabled
  onCheckedChange={() => {}}
/>
```

### Multiple Tiles

```tsx
import { CheckboxTile } from '@/components/ui/checkbox-tile';
import { View } from 'react-native';
import { useState } from 'react';

function MultipleCheckboxTiles() {
  const [option1, setOption1] = useState(false);
  const [option2, setOption2] = useState(false);
  const [option3, setOption3] = useState(false);

  return (
    <View className="gap-4">
      <CheckboxTile
        label="Option 1"
        description="Description for option 1"
        checked={option1}
        onCheckedChange={setOption1}
      />
      <CheckboxTile
        label="Option 2"
        description="Description for option 2"
        checked={option2}
        onCheckedChange={setOption2}
      />
      <CheckboxTile
        label="Option 3"
        description="Description for option 3"
        checked={option3}
        onCheckedChange={setOption3}
      />
    </View>
  );
}
```

## Helper Functions

### checkboxTileVariants

Use `checkboxTileVariants` to create custom checkbox tile styled components:

```tsx
import { checkboxTileVariants } from '@/components/ui/checkbox-tile';
import { View } from 'react-native';

<View className={checkboxTileVariants({ variant: 'withDescription', state: 'checked' })}>
  {/* content */}
</View>
```
