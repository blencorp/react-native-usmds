## Usage

```tsx
import { Checkbox } from '@/components/ui/checkbox';
import { useState } from 'react';

const [checked, setChecked] = useState(false);

<Checkbox
  checked={checked}
  onCheckedChange={setChecked}
/>
```

## Props

### Checkbox

Extends `@rn-primitives/checkbox` Root props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| checked | `boolean` | - | The controlled checked state. |
| onCheckedChange | `(checked: boolean) => void` | - | Callback when checked state changes. |
| disabled | `boolean` | `false` | Whether the checkbox is disabled. |
| className | `string` | - | Additional classes for the checkbox container. |
| checkedClassName | `string` | - | Additional classes when checked. |
| indicatorClassName | `string` | - | Additional classes for the indicator. |
| iconClassName | `string` | - | Additional classes for the check icon. |

### With Label

```tsx
import { Checkbox } from '@/components/ui/checkbox';

<View className="flex-row items-center gap-2">
  <Checkbox
    checked={checked}
    onCheckedChange={setChecked}
  />
  <Text>Accept terms and conditions</Text>

```

### Disabled

```tsx

```


### Custom Styling

```tsx

```

### Form Integration

```tsx
import { Checkbox } from '@/components/ui/checkbox';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';
import { useState } from 'react';

function FormExample() {
  const [acceptTerms, setAcceptTerms] = useState(false);
  const [receiveEmails, setReceiveEmails] = useState(false);

  return (
    <View className="gap-4">
      <View className="flex-row items-center gap-2">
        <Checkbox
          checked={acceptTerms}
          onCheckedChange={setAcceptTerms}
        />
        <Text>I accept the terms and conditions</Text>
      </View>

      <View className="flex-row items-center gap-2">
        <Checkbox
          checked={receiveEmails}
          onCheckedChange={setReceiveEmails}
        />
        <Text>Send me promotional emails</Text>
      </View>
    </View>
  );
}
```
