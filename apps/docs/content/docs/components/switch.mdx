## Usage

```tsx
import { Switch } from '@/components/ui/switch';
import { Label } from '@/components/ui/label';
import { useState } from 'react';

const [checked, setChecked] = useState(false);

<div className="flex items-center gap-2">
  <Switch
    checked={checked}
    onCheckedChange={setChecked}
    id="switch"
  />
  <Label htmlFor="switch">Label</Label>
</div>
```

## Props

### Switch

Extends `@rn-primitives/switch` Root props.

| Prop | Type | Description |
|------|------|-------------|
| checked | `boolean` | The controlled checked state of the switch. |
| onCheckedChange | `(checked: boolean) => void` | Event handler called when the checked state changes. |
| defaultChecked | `boolean` | The checked state when initially rendered. |
| disabled | `boolean` | When true, prevents the user from interacting with the switch. |
| className | `string` | Additional classes for styling. |

### Form Example

```tsx
import { Switch } from '@/components/ui/switch';
import { Button } from '@/components/ui/button';
import { Label } from '@/components/ui/label';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';
import { useState } from 'react';

function FormExample() {
  const [settings, setSettings] = useState({
    marketingEmails: false,
    socialEmails: true,
    securityEmails: true,
  });

  const handleSubmit = () => {
    console.log('Settings:', settings);
  };

  return (
    <View className="gap-4">
      <Text className="font-semibold">Email Notifications</Text>

      <div className="flex items-center justify-between">
        <Label htmlFor="marketing">Marketing emails</Label>
        <Switch
          id="marketing"
          checked={settings.marketingEmails}
          onCheckedChange={(checked) =>
            setSettings((prev) => ({ ...prev, marketingEmails: checked }))
          }
        />
      </div>

      <div className="flex items-center justify-between">
        <Label htmlFor="social">Social emails</Label>
        <Switch
          id="social"
          checked={settings.socialEmails}
          onCheckedChange={(checked) =>
            setSettings((prev) => ({ ...prev, socialEmails: checked }))
          }
        />
      </div>

      <div className="flex items-center justify-between">
        <Label htmlFor="security">Security emails</Label>
        <Switch
          id="security"
          checked={settings.securityEmails}
          onCheckedChange={(checked) =>
            setSettings((prev) => ({ ...prev, securityEmails: checked }))
          }
          disabled
        />
      </div>

      <Button onPress={handleSubmit}>
        <Text>Save preferences</Text>
      </Button>
    </View>
  );
}
```
