## Usage

```tsx
import { Progress } from '@/components/ui/progress';

<Progress value={40} />
```

## Props

### Progress

Extends `ProgressPrimitive.Root` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | `number` | - | The progress value between 0 and 100. |
| className | `string` | - | Additional classes for the root element. |
| indicatorClassName | `string` | - | Additional classes for the indicator element. |

### Animated Progress

```tsx
import { Progress } from '@/components/ui/progress';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { useState, useCallback } from 'react';
import { View } from 'react-native';

function AnimatedProgress() {
  const [progress, setProgress] = useState(40);

  const randomize = useCallback(() => {
    setProgress(Math.floor(Math.random() * 100));
  }, []);

  const increment = useCallback(() => {
    setProgress((prev) => Math.min(prev + 10, 100));
  }, []);

  return (
    <View className="gap-4">
      <Progress value={progress} />
      <Text>Progress: {progress}%</Text>
      <View className="flex-row gap-2">
        <Button onPress={increment}>
          <Text>Increment</Text>
        </Button>
        <Button onPress={randomize}>
          <Text>Randomize</Text>
        </Button>
      </View>
    </View>
  );
}
```

### Loading State

```tsx
function LoadingProgress() {
  const [progress, setProgress] = useState(0);

  useEffect(() => {
    const timer = setInterval(() => {
      setProgress((prev) => {
        if (prev >= 100) {
          clearInterval(timer);
          return 100;
        }
        return prev + 10;
      });
    }, 500);

    return () => clearInterval(timer);
  }, []);

  return <Progress value={progress} />;
}
```
