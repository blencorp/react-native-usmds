## Usage

```tsx
import { ButtonGroup } from '@/components/ui/button-group';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';

<ButtonGroup orientation="vertical">
  <Button variant="default">
    <Text>Primary</Text>
  </Button>
  <Button variant="outline">
    <Text>Secondary</Text>
  </Button>
</ButtonGroup>
```

## Props

### ButtonGroup

Extends `View` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| orientation | `'horizontal' \| 'vertical'` | `'vertical'` | The layout direction of the button group. |
| gap | `number` | `8` | The space between buttons in pixels. |
| className | `string` | - | Additional classes for styling. |

## Examples

### Vertical Layout

```tsx
import { ButtonGroup } from '@/components/ui/button-group';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';

<ButtonGroup orientation="vertical" gap={8}>
  <Button variant="default">
    <Text>Primary</Text>
  </Button>
  <Button variant="outline">
    <Text>Secondary</Text>
  </Button>
</ButtonGroup>
```

### Horizontal Layout

```tsx

  <Button variant="default">
    <Text>Primary</Text>
  </Button>

    <Text>Secondary</Text>

```

### Custom Gap

