## Usage

```tsx
import { Banner } from '@/components/ui/banner';

<Banner variant="standard-app" link="agency.gov" />
```

## Props

### Banner

Extends `View` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| variant | `'standard-app' \| 'welcome'` | `'standard-app'` | The visual style variant. |
| title | `string` | - | Custom title text (used with 'welcome' variant). |
| link | `string` | - | Domain link to display (used with 'standard-app' variant). |
| trailingIcon | `ReactNode` | - | Optional trailing icon element. |
| className | `string` | - | Additional classes for styling. |

## Variants

### Standard App

Shows the official US government app banner with flag icon and agency link.

```tsx
<Banner variant="standard-app" link="agency.gov" />
```

### Welcome

Displays a welcome banner with custom title text.

```tsx
<Banner variant="welcome" title="Welcome to our application" />
```

## Helper Functions

### bannerVariants

Use `bannerVariants` to create custom banner-styled components:

```tsx
import { bannerVariants } from '@/components/ui/banner';
import { View } from 'react-native';

<View className={bannerVariants({ variant: 'welcome' })}>
  {/* content */}
</View>
```

### bannerTextVariants

Use `bannerTextVariants` for text styling within banners:

```tsx
import { bannerTextVariants } from '@/components/ui/banner';
import { Text } from 'react-native';

<Text className={bannerTextVariants({ variant: 'standard-app' })}>
  Custom text
</Text>
```
