## Usage

```tsx
import { Pagination } from '@/components/ui/pagination';
import { useState } from 'react';

const [currentPage, setCurrentPage] = useState(1);

<Pagination
  currentPage={currentPage}
  totalPages={10}
  totalItems={100}
  onPageChange={setCurrentPage}
/>
```

## Props

### Pagination

Extends `View` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| currentPage | `number` | - | **Required.** The current active page number. |
| totalPages | `number` | - | **Required.** The total number of pages. |
| totalItems | `number` | - | **Required.** The total number of items across all pages. |
| onPageChange | `(page: number) => void` | - | **Required.** Callback fired when the page changes. |
| className | `string` | - | Additional classes for the container. |

### Basic Usage

```tsx

import { useState } from 'react';

function PaginationExample() {

```

### Large Dataset

```tsx

  const [currentPage, setCurrentPage] = useState(1);
  const totalPages = 100;
  const totalItems = 1000;

  return (
    <Pagination
      currentPage={currentPage}
      totalPages={totalPages}
      totalItems={totalItems}
      onPageChange={setCurrentPage}
    />
  );
}
```

### Controlled Pagination

```tsx
function ControlledPagination() {
  const [currentPage, setCurrentPage] = useState(1);
  const itemsPerPage = 10;
  const totalItems = 95;
  const totalPages = Math.ceil(totalItems / itemsPerPage);

  const handlePageChange = (page: number) => {
    setCurrentPage(page);
    // Fetch new data for this page
    fetchData(page, itemsPerPage);
  };

  return (
    <Pagination
      currentPage={currentPage}
      totalPages={totalPages}
      totalItems={totalItems}
      onPageChange={handlePageChange}
    />
  );
}
```
