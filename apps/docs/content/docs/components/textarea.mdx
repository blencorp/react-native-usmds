```tsx
  placeholder="Enter your text here..."
  value={text}

/>
```

## Props

### Textarea

Extends React Native's `TextInput` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| className | `string` | - | Additional classes for styling. |
| placeholderClassName | `string` | - | Additional classes for placeholder text. |
| multiline | `boolean` | `true` | Enables multiline input. |
| numberOfLines | `number` | `8` (native), `2` (web) | Number of lines to display. |
| editable | `boolean` | `true` | When false, textarea is disabled. |
| value | `string` | - | The textarea value. |
| onChangeText | `(text: string) => void` | - | Callback when text changes. |

All standard React Native TextInput props are supported.

### With Error State

```tsx
import { Textarea } from '@/components/ui/textarea';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';
import { useState } from 'react';

function TextareaWithError() {
  const [text, setText] = useState('');
  const [error, setError] = useState('');

  const validateText = (value: string) => {
    setText(value);
    if (value.length < 10) {
      setError('Please enter at least 10 characters');
    } else {
      setError('');
    }
  };

  return (
    <View className="gap-2">
      <Textarea
        placeholder="Enter your feedback..."
        value={text}
        onChangeText={validateText}
        aria-invalid={!!error}
        className={error ? 'border-destructive' : ''}
      />
      {error && <Text className="text-sm text-destructive">{error}</Text>}
    </View>
  );
}
```

### Form Example

```tsx
import { Textarea } from '@/components/ui/textarea';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';
import { useState } from 'react';

function FeedbackForm() {
  const [feedback, setFeedback] = useState('');

  const handleSubmit = () => {
    console.log('Feedback:', feedback);
    setFeedback('');
  };

  return (
    <View className="gap-4">
      <View className="gap-2">
        <Text className="font-semibold">Send us your feedback</Text>
        <Textarea
          placeholder="Tell us what you think..."
          value={feedback}
          onChangeText={setFeedback}
        />
      </View>

      <Button onPress={handleSubmit} disabled={!feedback}>
        <Text>Submit Feedback</Text>
      </Button>
    </View>
  );
}
```

### Comment Box

```tsx
import { Textarea } from '@/components/ui/textarea';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';
import { useState } from 'react';

function CommentBox() {
  const [comment, setComment] = useState('');

  const handlePost = () => {
    if (comment.trim()) {
      console.log('Posting comment:', comment);
      setComment('');
    }
  };

  return (
    <View className="gap-3 p-4 border rounded-lg">
      <Text className="font-semibold">Add a comment</Text>
      <Textarea
        placeholder="Write your comment here..."
        value={comment}
        onChangeText={setComment}
        className="min-h-24"
      />
      <View className="flex-row justify-end gap-2">
        <Button variant="ghost" onPress={() => setComment('')}>
          <Text>Cancel</Text>
        </Button>
        <Button onPress={handlePost} disabled={!comment.trim()}>
          <Text>Post Comment</Text>
        </Button>
      </View>
    </View>
  );
}
```
