## Usage

```tsx
import {
  Card,
  CardHeader,
  CardTitle,
  CardDescription,
  CardContent,
  CardFooter,
} from '@/components/ui/card';
import { Text } from '@/components/ui/text';

<Card>
  <CardHeader>
    <CardTitle>Card Title</CardTitle>
    <CardDescription>Card description</CardDescription>
  </CardHeader>
  <CardContent>
    <Text>Card content</Text>
  </CardContent>
  <CardFooter>
    {/* Footer content */}
  </CardFooter>
</Card>
```

## Examples

### Default

```tsx
import { Card, CardHeader, CardTitle, CardDescription, CardContent } from '@/components/ui/card';
import { Text } from '@/components/ui/text';

<Card>
  <CardHeader>
    <CardTitle>Notifications</CardTitle>
    <CardDescription>You have 3 unread messages</CardDescription>
  </CardHeader>
  <CardContent>
    <Text>Message content here</Text>
  </CardContent>
</Card>
```

### With Footer

```tsx
import {
  Card,
  CardHeader,
  CardTitle,
  CardDescription,
  CardContent,
  CardFooter,
} from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';

<Card>
  <CardHeader>
    <CardTitle>Create Account</CardTitle>
    <CardDescription>Enter your information below</CardDescription>
  </CardHeader>
  <CardContent>
    <Text>Form fields here</Text>
  </CardContent>
  <CardFooter>
    <Button variant="default">
      <Text>Submit</Text>
    </Button>
  </CardFooter>
</Card>
```

### Custom Styling

```tsx
<Card className="border-2 border-primary">
  <CardHeader>
    <CardTitle className="text-primary">Featured</CardTitle>
    <CardDescription>This is a featured card</CardDescription>
  </CardHeader>
  <CardContent>
    <Text>Special content</Text>
  </CardContent>
</Card>
```

### With Image

```tsx
import { Card, CardHeader, CardTitle, CardDescription, CardContent } from '@/components/ui/card';
import { Image, Text } from 'react-native';

<Card>
  <Image
    source={{ uri: 'https://example.com/image.jpg' }}
    className="w-full h-48 rounded-t-xl"
  />
  <CardHeader>
    <CardTitle>Image Card</CardTitle>
    <CardDescription>Card with image on top</CardDescription>
  </CardHeader>
  <CardContent>
    <Text>Content below the image</Text>
  </CardContent>
</Card>
```
