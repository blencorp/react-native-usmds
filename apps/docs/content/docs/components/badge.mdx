## Usage

```tsx
import { Badge } from '@/components/ui/badge';
import { Text } from '@/components/ui/text';

<Badge variant="default">
  <Text>Badge</Text>
</Badge>
```

## Props

### Badge

Extends `View` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| variant | `'default' \| 'secondary' \| 'destructive' \| 'outline'` | `'default'` | The visual style variant. |
| asChild | `boolean` | `false` | Render as a child component (uses Slot). |
| className | `string` | - | Additional classes for styling. |

## Variants

### Default

```tsx
<Badge variant="default">
  <Text>Default</Text>
</Badge>
```

### Secondary

```tsx
<Badge variant="secondary">
  <Text>Secondary</Text>
</Badge>
```

### Destructive

```tsx
<Badge variant="destructive">
  <Text>Destructive</Text>
</Badge>
```

### Outline

```tsx
<Badge variant="outline">
  <Text>Outline</Text>
</Badge>
```

## Helper Functions

### badgeVariants

Use `badgeVariants` to create badge-styled components:

```tsx
import { badgeVariants } from '@/components/ui/badge';
import { View } from 'react-native';

<View className={badgeVariants({ variant: 'outline' })}>
  {/* content */}
</View>
```

### badgeTextVariants

Use `badgeTextVariants` for text styling within badges:

```tsx
import { badgeTextVariants } from '@/components/ui/badge';
import { Text } from '@/components/ui/text';

<Text className={badgeTextVariants({ variant: 'destructive' })}>
  Error
</Text>
```

## Examples

### With Icon

```tsx
import { Badge } from '@/components/ui/badge';
import { Text } from '@/components/ui/text';
import { Check } from 'lucide-react-native';

<Badge variant="default">
  <Check />
  <Text>Verified</Text>
</Badge>
```

### As Link

```tsx
import { badgeVariants, badgeTextVariants } from '@/components/ui/badge';
import { Text } from '@/components/ui/text';
import { Link } from 'expo-router';

<Link href="/profile" className={badgeVariants({ variant: 'outline' })}>
  <Text className={badgeTextVariants({ variant: 'outline' })}>
    View Profile
  </Text>
</Link>
```

### Multiple Badges

```tsx
import { Badge } from '@/components/ui/badge';
import { Text } from '@/components/ui/text';
import { View } from 'react-native';

<View className="flex-row gap-2">
  <Badge variant="default">
    <Text>New</Text>
  </Badge>
  <Badge variant="secondary">
    <Text>Featured</Text>
  </Badge>
  <Badge variant="outline">
    <Text>Sale</Text>
  </Badge>
</View>
```
