## Usage

```tsx
import { StepIndicator } from '@/components/ui/stepindicator';

<StepIndicator
  steps={6}
  currentStep={3}
  title="Current Step Title"
  size="default"
/>
```

## Props

### StepIndicator

Extends `View` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| steps | `number` | - | **Required.** Total number of steps in the sequence. |
| currentStep | `number` | - | **Required.** The current active step number (1-indexed). |
| title | `string` | - | **Required.** Title text for the current step. |
| size | `'default' \| 'small'` | `'default'` | The size variant of the step indicator. |
| labels | `string[]` | - | Optional array of labels for each step. |
| className | `string` | - | Additional classes for styling. |

### Multi-Step Form

```tsx
import { StepIndicator } from '@/components/ui/stepindicator';
import { Button } from '@/components/ui/button';
import { Text } from '@/components/ui/text';
import { useState } from 'react';
import { View } from 'react-native';

function MultiStepForm() {
  const [currentStep, setCurrentStep] = useState(1);
  const totalSteps = 5;

  const stepTitles = [
    'Personal Info',
    'Contact Details',
    'Address',
    'Payment',
    'Review'
  ];

  const handleNext = () => {
    if (currentStep < totalSteps) {
      setCurrentStep(currentStep + 1);
    }
  };

  const handlePrevious = () => {
    if (currentStep > 1) {
      setCurrentStep(currentStep - 1);
    }
  };

  return (
    <View className="gap-4">
      <StepIndicator
        steps={totalSteps}
        currentStep={currentStep}
        title={stepTitles[currentStep - 1]}
      />

      <View className="flex-row gap-2">
        <Button
          onPress={handlePrevious}
          disabled={currentStep === 1}
        >
          <Text>Previous</Text>
        </Button>
        <Button
          onPress={handleNext}
          disabled={currentStep === totalSteps}
        >
          <Text>Next</Text>
        </Button>
      </View>
    </View>
  );
}
```

### Progress Tracking

```tsx
function ProgressTracking() {
  const [progress, setProgress] = useState(1);
  const totalSteps = 8;

  const handleComplete = () => {
    if (progress < totalSteps) {
      setProgress(progress + 1);
    }
  };

  return (
    <View className="gap-4">
      <StepIndicator
        steps={totalSteps}
        currentStep={progress}
        title={`Step ${progress}: In Progress`}
        size="default"
      />

      <Text className="text-sm text-muted-foreground">
        {Math.round((progress / totalSteps) * 100)}% Complete
      </Text>

      <Button onPress={handleComplete} disabled={progress === totalSteps}>
        <Text>Complete Step</Text>
      </Button>
    </View>
  );
}
```
