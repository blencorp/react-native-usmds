## Usage

```tsx
import { Toggle, ToggleIcon } from '@/components/ui/toggle';
import { Bold } from 'lucide-react-native';
import { useState } from 'react';

const [pressed, setPressed] = useState(false);

<Toggle pressed={pressed} onPressedChange={setPressed}>
  <ToggleIcon as={Bold} />
</Toggle>
```

## Props

### Toggle

Extends `TogglePrimitive.Root` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| pressed | `boolean` | `false` | The controlled pressed state of the toggle. |
| onPressedChange | `(pressed: boolean) => void` | - | Callback when pressed state changes. |
| variant | `'default' \| 'outline'` | `'default'` | The visual style variant. |
| size | `'default' \| 'sm' \| 'lg'` | `'default'` | The size of the toggle. |
| disabled | `boolean` | `false` | When true, prevents user interaction. |
| className | `string` | - | Additional classes for styling. |

### ToggleIcon

Extends `Icon` props.

| Prop | Type | Description |
|------|------|-------------|
| as | `LucideIcon` | The icon component to display. |
| className | `string` | Additional classes for styling. |

```tsx
import { Toggle, ToggleIcon } from '@/components/ui/toggle';

  const [bold, setBold] = useState(false);
  const [italic, setItalic] = useState(false);
  const [underline, setUnderline] = useState(false);

  return (
    <View className="flex-row gap-1">
      <Toggle pressed={bold} onPressedChange={setBold} variant="outline" size="sm">
        <ToggleIcon as={Bold} />
      </Toggle>
      <Toggle pressed={italic} onPressedChange={setItalic} variant="outline" size="sm">
        <ToggleIcon as={Italic} />
      </Toggle>
      <Toggle pressed={underline} onPressedChange={setUnderline} variant="outline" size="sm">
        <ToggleIcon as={Underline} />
      </Toggle>
    </View>
  );
}
```

### Single Icon Toggle

```tsx
import { Toggle, ToggleIcon } from '@/components/ui/toggle';
import { Volume2, VolumeX } from 'lucide-react-native';
import { useState } from 'react';

function VolumeToggle() {
  const [muted, setMuted] = useState(false);

  return (
    <Toggle pressed={muted} onPressedChange={setMuted} variant="outline">
      <ToggleIcon as={muted ? VolumeX : Volume2} />
    </Toggle>
  );
}
```

## Helper Functions

### toggleVariants

Use `toggleVariants` to create custom toggle-styled components:

```tsx
import { toggleVariants } from '@/components/ui/toggle';
import { Pressable } from 'react-native';

<Pressable className={toggleVariants({ variant: 'outline', size: 'lg' })}>
  {/* custom content */}
</Pressable>
```
